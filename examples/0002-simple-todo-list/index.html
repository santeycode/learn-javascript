Добавление задач в список

Пример

Список задач

    HTML

    <h1>Список задач</h1> <ul id="list"></ul> <input type="text" id="inputTask" placeholder="Название задачи"> <button id="addTask">Добавить задачу</button>

    JavaScript

    var list = document.getElementById('list'); var input = document.getElementById('inputTask'); var button = document.getElementById('addTask'); function addTask() { if (input.value !== '') { list.innerHTML += '<li class="list-group-item">' + input.value + '</li>'; input.value = ''; } } button.onclick = addTask;
